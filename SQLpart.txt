    name TEXT,
    birth_date DATE,

);

    id INTEGER PRIMARY KEY,
    name TEXT,
    birth_date DATE,
    commands TEXT
);

CREATE TABLE hamsters (
    id INTEGER PRIMARY KEY,
    name TEXT,
    birth_date DATE,
    commands TEXT
);

CREATE TABLE horses (
    id INTEGER PRIMARY KEY,
    name TEXT,
    birth_date DATE,
    commands TEXT
);

CREATE TABLE camels (
    id INTEGER PRIMARY KEY,
    name TEXT,
    birth_date DATE,
    commands TEXT
);

CREATE TABLE donkeys (
    id INTEGER PRIMARY KEY,
    name TEXT,
    birth_date DATE,
    commands TEXT

-- Объединение таблиц лошадей и ослов
CREATE TABLE horses_and_donkeys AS
SELECT * FROM horses
UNION ALL
SELECT * FROM donkeys;

-- Создание новой таблицы "молодые животные"
CREATE TABLE young_animals AS
SELECT *, TIMESTAMPDIFF(MONTH, birth_date, CURDATE()) AS age_in_months
FROM (
    SELECT * FROM dogs
    UNION
    SELECT * FROM cats
    UNION
    SELECT * FROM hamsters
    UNION
    SELECT * FROM horses_and_donkeys
) AS all_animals
WHERE birth_date >= CURDATE() - INTERVAL 3 YEAR AND birth_date <= CURDATE() - INTERVAL 1 YEAR;

-- Объединение всех таблиц в одну
CREATE TABLE all_animals AS
SELECT *, 'dogs' AS source_table FROM dogs
UNION ALL
SELECT *, 'cats' AS source_table FROM cats
UNION ALL
SELECT *, 'hamsters' AS source_table FROM hamsters
UNION ALL
SELECT *, 'horses' AS source_table FROM horses
UNION ALL
SELECT *, 'camels' AS source_table FROM camels
UNION ALL
SELECT *, 'donkeys' AS source_table FROM donkeys;
